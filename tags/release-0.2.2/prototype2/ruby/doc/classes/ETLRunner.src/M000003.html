  def init_chain(chain_config, chain_name)
    chain = ChainETL.new
    chain.phase = chain_name.to_sym
    i = 0
    while token = chain_config["block#{i}"] do
      etlclass , etlname = token.split(",").reject{ |x| x == '' }
      etl = eval("#{etlclass}.new")
      etl.name = etlname || "#{chain_name}_#{i}_#{etlclass}"
      etl.props = @opts[etl.name]
      chain << etl
      i += 1
    end
    return chain 
  end
