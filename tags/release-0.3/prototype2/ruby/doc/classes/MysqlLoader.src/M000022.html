  def load(dto,context)
    print_warnings(dto)
    begin
      open_connection
      
      with_source(dto) do |source_id|
        with_page(dto,source_id) do |page_id|
          insert_words(source_id,page_id,dto)
        end
      end
      
    rescue Mysql::Error => me
      log "Mysql Error: #{me}"
      raise me
    rescue Exception => e
      log "Generic Error: #{e}"
      raise e
    ensure
      close_connection
    end
  end
