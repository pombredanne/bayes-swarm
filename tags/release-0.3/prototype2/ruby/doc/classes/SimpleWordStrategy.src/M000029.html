  def insert(source_id,page_id,scantime,word)
    @page_id , @scantime , @word = page_id, scantime, word

    insert_if_needed
    
    case word.position
    when "global"
      update "count"
    when "title"
      update "titlecount"
    else
      puts "Unrecognized word position #{word.position} for word #{word.word} and page #{page_id}"
    end
  end
