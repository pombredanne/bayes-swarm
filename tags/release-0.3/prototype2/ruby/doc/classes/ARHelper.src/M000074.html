  def with_connection(params={})
    begin
      open_connection(params)
      conn = ActiveRecord::Base.connection
      yield conn
    rescue ActiveRecordError => e
      log "ActiveRecord Error: #{e}"
      raise e
    rescue Exception => e
      log "Generic Error: #{e}"
      raise e
    ensure
      close_connection
    end
  end
