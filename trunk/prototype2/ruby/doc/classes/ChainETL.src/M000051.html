  def unless_step_over(block, ctx)
    if ctx[:initialstep].nil? or block.name == ctx[:initialstep]
      verbose_log "Invoking phase #{@phase} on #{block.name}"
      ctx[:initialstep] = nil
      
      yield(block, ctx)
      
    else
      ctx[:curstep] = block.name
      verbose_log "Skipping block #{block.name}"
    end
  end
