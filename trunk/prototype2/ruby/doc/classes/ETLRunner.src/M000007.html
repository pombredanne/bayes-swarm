  def run(extra_context = {})
    @context = init_context()    
    enrich_context(extra_context)
    enable_savepoints if global_opt(:savepoint)  

    begin
      @root_chain.run_chain(@context[:dto],@context) { |block| block.name.to_sym }
      clean
    rescue Exception => e
    end
  end
